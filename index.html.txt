  <script>
    // Ripple
    document.querySelectorAll('[data-ripple]').forEach(btn => {
      btn.addEventListener('click', (e) => {
        const rect = btn.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;

        const ripple = document.createElement('span');
        ripple.className = 'ripple';
        ripple.style.left = x + 'px';
        ripple.style.top = y + 'px';
        ripple.style.width = ripple.style.height = Math.max(rect.width, rect.height) + 'px';

        btn.appendChild(ripple);
        ripple.addEventListener('animationend', () => ripple.remove());
      }, {passive:true});
    });

    // Parallax background (gentle)
    const bg = document.getElementById('bg');
    let targetX = 0, targetY = 0, curX = 0, curY = 0;

    function onMove(e){
      const x = (e.touches ? e.touches[0].clientX : e.clientX) / innerWidth - 0.5;
      const y = (e.touches ? e.touches[0].clientY : e.clientY) / innerHeight - 0.5;
      targetX = x * 14;
      targetY = y * 14;
    }
    addEventListener('mousemove', onMove, {passive:true});
    addEventListener('touchmove', onMove, {passive:true});

    function parallaxTick(){
      curX += (targetX - curX) * 0.06;
      curY += (targetY - curY) * 0.06;
      bg.style.transform = `translate3d(${curX}px, ${curY}px, 0) scale(1.03)`;
      requestAnimationFrame(parallaxTick);
    }
    parallaxTick();

    // Particles (minimal & light)
    const canvas = document.getElementById('fx');
    const ctx = canvas.getContext('2d');
    let w, h, dpr, particles;

    function resize(){
      dpr = Math.max(1, Math.min(2, window.devicePixelRatio || 1));
      w = canvas.width = Math.floor(innerWidth * dpr);
      h = canvas.height = Math.floor(innerHeight * dpr);
      canvas.style.width = innerWidth + 'px';
      canvas.style.height = innerHeight + 'px';

      const count = Math.floor((innerWidth * innerHeight) / 19000);
      particles = Array.from({length: Math.max(28, Math.min(80, count))}, () => ({
        x: Math.random() * w,
        y: Math.random() * h,
        r: (Math.random() * 1.6 + 0.6) * dpr,
        vx: (Math.random() * 0.42 + 0.08) * dpr,
        vy: (Math.random() * 0.22 + 0.05) * dpr,
        a: Math.random() * 0.32 + 0.08
      }));
    }

    function tick(){
      ctx.clearRect(0,0,w,h);

      particles.forEach(p => {
        p.x += p.vx; p.y += p.vy;
        if(p.x > w + 40) p.x = -40;
        if(p.y > h + 40) p.y = -40;

        ctx.beginPath();
        ctx.fillStyle = `rgba(243,194,139,${p.a})`;
        ctx.arc(p.x, p.y, p.r, 0, Math.PI*2);
        ctx.fill();
      });

      requestAnimationFrame(tick);
    }

    addEventListener('resize', resize);
    resize(); tick();
  </script>
